1   package fukroglyk.logiikka.maailma;
2   
3   import fukroglyk.entiteetit.Pelaaja;
4   import fukroglyk.entiteetit.Piirrettava;
5   import java.util.ArrayList;
6   
7   /**
8    * Luokka luo sekä hallinnoi pelin maailmassa olevia alueita
9    *
10   * @author NukeCow
11   */
12  public class Maailma {
13  
14      private int currentzone;
15      private ArrayList<Alue> alueet;
16      private Pelaaja pelaaja;
17  
18      public Maailma(Pelaaja pelaaja) {
19          this.pelaaja = pelaaja;
20          this.currentzone = 0;
21          this.alueet = new ArrayList();
22      }
23  
24      /**
25       * Luo tämänhetkisen maailman kaikki alueet.
26       */
27      public void luoMaailma() {
28          luoTaverna();
29          luoNiitty();
30      }
31  
32      /**
33       * Ennenkuin alueiden tieto luetaan tiedostoista, luo pelin ensimmäisen
34       * alueen ja lisää sen listaan alueista.
35       */
36      public void luoTaverna() {
37          int[] x = new int[]{13, 14, 15, 2, 13, 4, 14, 12, 13, 15, 4, 9, 13};
38          int[] y = new int[]{2, 5, 7, 9, 10, 12, 12, 15, 15, 15, 17, 18, 18};
39          Alue taverna = new Alue(0, "Taverna", x, y, 2, 2);
40          taverna.generoiLaatat();
41          this.alueet.add(taverna);
42      }
43  
44      /**
45       * Ennenkuin alueiden tieto luetaan tiedostoista, luo pelin toisen alueen ja
46       * lisää sen listaan alueista.
47       */
48      public void luoNiitty() {
49          int[] x = new int[]{5, 7, 6};
50          int[] y = new int[]{3, 8, 2};
51          Alue niitty = new Alue(1, "Niitty", x, y, 1, 19);
52          niitty.generoiLaatat();
53          this.alueet.add(niitty);
54      }
55  
56      /**
57       * Palauttaa pelin aktiivisen alueen id:n.
58       *
59       * @return pelin aktiivisen alueen id
60       */
61      public int getCurrentZone() {
62          return this.currentzone;
63      }
64  
65      /**
66       * Palauttaa pelin viimeisen mahdollisen alueen id:n.
67       *
68       * @return
69       */
70      public int getMaxZone() {
71          return this.alueet.size() - 1;
72      }
73      
74      public int getAlueidenMaara() {
75          return this.alueet.size();
76      }
77  
78      /**
79       * Asettaa seuraavan alueen aktiiviseksi alueeksi jos tämänhetkinen
80       * aktiivinen alue ei ole viimeinen mahdollinen alue.
81       *
82       * @return tieto siitä muutettiinko aluetta vai ei
83       */
84      public boolean setNextZone() {
85          if (this.currentzone < getMaxZone()) {
86              this.currentzone++;
87              return true;
88          }
89          return false;
90      }
91  
92      /**
93       * Asettaa edellisen alueen aktiiviseksi alueeksi jos tämänhetkinen
94       * aktiivinen alue ei ole aloitusalue.
95       *
96       * @return tieto siitä muutettiinko aluetta vai ei
97       */
98      public boolean setPrevZone() {
99          if (this.currentzone > 0) {
100             this.currentzone--;
101             return true;
102         }
103         return false;
104     }
105 
106     /**
107      * Palauttaa alueen jolla pelaaja tällähetkellä on.
108      *
109      * @return Aktiivinen alue
110      */
111     public Alue getCurrentAlue() {
112         return this.alueet.get(this.currentzone);
113     }
114 
115     /**
116      * Palauttaa aktiivisen alueen laattatiedot.
117      *
118      * @return Lista alueen laatoista
119      */
120     public ArrayList<Laatta> getCurrentLaatat() {
121         return getCurrentAlue().getLaatat();
122     }
123 
124     /**
125      * Palauttaa aktiivisen alueen piirrettävät oliot.
126      *
127      * @return Lista piirrettävistä olioista
128      */
129     public ArrayList<Piirrettava> getPiirrettavat() {
130         return getCurrentAlue().getPiirrettavat();
131     }
132 
133 }
