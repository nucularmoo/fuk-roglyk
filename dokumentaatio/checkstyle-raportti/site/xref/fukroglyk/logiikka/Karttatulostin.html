3   import fukroglyk.entiteetit.Hahmo;
4   import fukroglyk.entiteetit.Tavara;
5   import java.util.ArrayList;

7   public class Karttatulostin {
8   
9       private char[][] kartta;
10      private ArrayList<Hahmo> hahmot;
11      private ArrayList<Tavara> tavarat;
12  
13      public Karttatulostin(int x, int y) {
14  
15          this.kartta = new char[y + 1][x + 1];
16  
17      }
18  
19      public void hahmoLista(ArrayList<Hahmo> hahmot) {
20          this.hahmot = hahmot;
21      }
22  
23      public void tavaraLista(ArrayList<Tavara> tavarat) {
24          this.tavarat = tavarat;
25      }
26  
27      //Kartan päivitys
28      public void update() {
29          tyhjaKartta();
30          lisaaTavarat();
31          lisaaHahmot();
32      }
33  
34      //Alustetaan karttamatriisi
35      public void tyhjaKartta() {
36          for (int i = 0; i < this.kartta.length; i++) {
37              for (int j = 0; j < this.kartta[i].length; j++) {
38                  this.kartta[i][j] = '.';
39              }
40          }
41      }
42  
43      //Lisätään karttamatriisiin tavarat
44      public void lisaaTavarat() {
45          for (Tavara tavara : this.tavarat) {
46              if (!tavara.poimittu()) {
47                  this.kartta[tavara.getY()][tavara.getX()] = tavara.getChar();
48              }
49          }
50      }
51  
52      //Lisätään karttamatriisiin hahmot
53      public void lisaaHahmot() {
54          for (Hahmo hahmo : this.hahmot) {
55              this.kartta[hahmo.getY()][hahmo.getX()] = hahmo.getChar();
56          }
57      }
58  
59      //Kartan konsolituloste
60      public void print() {
61          for (int i = 0; i < this.kartta.length; i++) {
62              for (int j = 0; j < this.kartta[i].length; j++) {
63                  System.out.print(this.kartta[i][j]);
64              }
65              System.out.println("");
66          }
67      }
68  }
