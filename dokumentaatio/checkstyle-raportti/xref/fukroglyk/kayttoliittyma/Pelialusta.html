3   import fukroglyk.logiikka.Peli;
4   import fukroglyk.tiedostohallinta.Kuvanlukija;
5   import java.awt.BorderLayout;
6   import javax.swing.JFrame;
7   import java.awt.Canvas;
8   import java.awt.Container;
9   import java.awt.Dimension;
10  import java.awt.image.BufferedImage;
11  import javax.swing.ImageIcon;
12  import javax.swing.JLabel;
13  
14  /**
15   * Luokka luo ja hallinnoi graafista käyttöliittymää.
16   *
17   * @author NukeCow
18   */
19  public class Pelialusta extends Canvas implements Runnable {
20  
21      public Aloituskuuntelija derp;
22  
23      // ikkunan kokotiedot ja title
24      public static final int WIDTH = 160;
25      public static final int HEIGHT = WIDTH;
26      public static final int SCALE = 3;
27      public static final String NAME = "fuk-roglyk";
28  
29      private JFrame frame;
30  
31      private Piirtoalusta alusta;
32  
33      //peli käynnissä vai ei
34      public boolean kaynnissa = false;
35  
36      public Peli peli;
37  
38      public Piirtoalusta piirtoalusta;
39  
40      public Kuvanlukija lukija;
41      public BufferedImage image;
42  
43      public boolean start = false;
44  
45      public Pelialusta() {
46          this.peli = new Peli();
47          this.peli.init();
48          this.peli.luoMaailmanHallinta();
49          this.peli.luoMaailma();
50      }
51  
52      public void haeKuva() {
53          this.lukija = new Kuvanlukija();
54          this.lukija.haeKuva();
55          this.image = this.lukija.getKuva();
56      }
57  
58      /**
59       * Alustaa ikkunan tiedot ja rakentaa kehyksen jossa pelin graafinen
60       * toteutus tapahtuu.
61       */
62      public void init() {
63          this.piirtoalusta = new Piirtoalusta(peli.getPiirrettavat());
64          haeKuva();
65          asetaIkkunanKoko();
66          rakennaKehys();
67      }
68  
69      /**
70       * Asettaa pelin ikkunan maksimi-, minimi- ja suosituskoon mitat.
71       */
72      public void asetaIkkunanKoko() {
73          // asetetaan ikkunan koko
74          setMinimumSize(new Dimension(WIDTH * SCALE, HEIGHT * SCALE));
75          setMaximumSize(new Dimension(WIDTH * SCALE, HEIGHT * SCALE));
76          setPreferredSize(new Dimension(WIDTH * SCALE, HEIGHT * SCALE));
77      }
78  
79      /**
80       * Rakentaa pelin ikkunan ja tekee siitä näkyvän.
81       */
82      public void rakennaKehys() {
83          //luodaan ikkuna ja sen sulkeutuminen
84          if (!this.start) {
85              frame = new JFrame(NAME);
86              frame.setPreferredSize(new Dimension(WIDTH * SCALE, HEIGHT * SCALE));
87              frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
88  
89              //luodaan ikkunan layout
90              frame.setLayout(new BorderLayout());
91              frame.add(this, BorderLayout.CENTER);
92              luoAloitusIkkuna(frame.getContentPane());
93              frame.pack();
94              frame.setResizable(false);
95              frame.setLocationRelativeTo(null);
96              frame.setVisible(true);
97              this.start = true;
98          } else {
99              frame.getContentPane().removeAll();
100             frame.removeKeyListener(derp);
101             luoKomponentit(frame.getContentPane());
102             frame.pack();
103             frame.setResizable(false);
104             frame.setVisible(true);
105         }
106     }
107 
108     /**
109      * Luo komponentit aloitusikkunalle.
110      *
111      * @param container
112      */
113     public void luoAloitusIkkuna(Container container) {
114         JLabel label = new JLabel(new ImageIcon(this.image));
115         container.add(label);
116         this.derp = new Aloituskuuntelija(this);
117         frame.addKeyListener(this.derp);
118     }
119 
120     /**
121      * Luo ikkunan komponentit piirrettävien piirtämiselle sekä
122      * näppäimistökomentojen kuuntelemiselle.
123      *
124      * @param container
125      */
126     public void luoKomponentit(Container container) {
127         container.add(this.piirtoalusta);
128         Nappaimistonkuuntelija kuuntelija = new Nappaimistonkuuntelija(peli.getLH(), this.piirtoalusta);
129         frame.addKeyListener(kuuntelija);
130     }
131 
132     /**
133      * Käynnistää pelin
134      */
135     public synchronized void kaynnista() {
136         init();
137         kaynnissa = true;
138         new Thread(this).start();
139         while (kaynnissa) {
140             peli.piirra();
141             this.piirtoalusta.setPiirrettavat(peli.getPiirrettavat());
142         }
143     }
144 
145     /**
146      * Sammuttaa pelin.
147      */
148     public synchronized void sammuta() {
149         kaynnissa = false;
150     }
151 
152     @Override
153     public void run() {
154 
155     }
156 
157 }
