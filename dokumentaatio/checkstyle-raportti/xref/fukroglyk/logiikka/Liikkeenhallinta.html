8   /**
9    * Luokka hallinnoi pelaajan liikkumista ja vuorovaikuttamista maailmassa.
10   *
11   * @author NukeCow
12   */
13  public class Liikkeenhallinta {
14  
15      private ArrayList<Laatta> laatat;
16      private Peli peli;
17      private Pelaaja pelaaja;
18      private Alue alue;
19      private boolean playerSet;
20      private Tapahtumankasittelija tk;
21  
22      public Liikkeenhallinta() {
23  
24      }
25  
26      public Liikkeenhallinta(Peli peli) {
27          this.playerSet = false;
28          this.peli = peli;
29          this.alue = this.peli.getCurrentAlue();
30          this.laatat = new ArrayList();
31      }
32  
33      /**
34       * Asettaa liikkeenhallinnalle tapahtumakäsittelijän.
35       *
36       * @param tk pelin tapahtumakäsittelijä
37       */
38      public void setTapahtumankasittelija(Tapahtumankasittelija tk) {
39          this.tk = tk;
40      }
41  
42      public void setAlue(Alue alue) {
43          this.alue = alue;
44      }
45  
46      public Alue getAlue() {
47          return this.alue;
48      }
49  
50      public void setLaatat(ArrayList<Laatta> laatat) {
51          this.laatat = laatat;
52      }
53  
54      public void setPelaaja(Pelaaja pelaaja) {
55          this.pelaaja = pelaaja;
56      }
57  
58      public boolean getPlayerSet() {
59          return this.playerSet;
60      }
61  
62      /**
63       * Liikkeenhallinta on asettanut pelaajan aloituspaikalleen.
64       */
65      public void setPlayerSet() {
66          this.playerSet = true;
67      }
68  
69      /**
70       * Liikkeenhallinnan käynnistyessä varmistetaan että pelaaja on oikeassa
71       * paikassa.
72       */
73      public void init() {
74          if (!getPlayerSet()) {
75              validoiPelaaja(this.pelaaja);
76          }
77      }
78  
79      /**
80       * Varmistetaan että pelaaja ei vahingossa ilmene maailmaan alueen
81       * ulkopuoliselle alueelle tai laatalle jossa on jo jotakin.
82       *
83       * @param pelaaja pelin pelaaja
84       */
85      public void validoiPelaaja(Pelaaja pelaaja) {
86          int px = pelaaja.getX();
87          int py = pelaaja.getY();
88  
89          if (lowerBoundsCheck(px, py)) {
90              resetPlayerByAlue();
91          } else if (upperBoundsCheck(px, py)) {
92              resetPlayerByAlue();
93          } else if (collision(px, py)) {
94              resetPlayerByAlue();
95          }
96          setPlayerSet();
97      }
98  
99      /**
100      * Laskee x-koordinaatin johon pelaaja yrittää siirtyä.
101      *
102      * @param xa haluttu muutos x-koordinaattiin
103      * @return x-koordinaatti johon pelaaja yrittää siirtyä
104      */
105     public int laskePelaajanUusiX(int xa) {
106         return this.pelaaja.getX() + xa;
107     }
108 
109     /**
110      * Laskee y-koordinaatin johon pelaaja yrittää siirtyä.
111      *
112      * @param ya haluttu muutos y-koordinaattiin
113      * @return y-koordinaatti johon pelaaja yrittää siirtyä
114      */
115     public int laskePelaajanUusiY(int ya) {
116         return this.pelaaja.getY() + ya;
117     }
118 
119     /**
120      * Resetoi pelaajan koordinaatit alueen antaman tiedon mukaisesti.
121      */
122     public void resetPlayerByAlue() {
123         this.pelaaja.setX(this.alue.getAloitusX());
124         this.pelaaja.setY(this.alue.getAloitusY());
125     }
126 
127     /**
128      * Metodi liikuttaa pelaajaa annettujen arvojen mukaisesti jos pelaaja on
129      * liikkumassa koordinaattiin johon pelaaja voi liikkua.
130      *
131      * @param xa haluttu muutos x-koordinaattiin
132      * @param ya haluttu muutos y-koordinaattiin
133      */
134     public void move(int xa, int ya) {
135         int nx = laskePelaajanUusiX(xa);
136         int ny = laskePelaajanUusiY(ya);
137         if (!collision(nx, ny)) {
138             this.pelaaja.moveX(xa);
139             this.pelaaja.moveY(ya);
140         }
141     }
142 
143     /**
144      * Varmistaa että halutut koordinaatit ovat suuremmat kuin niiden pienin
145      * mahdollinen arvo.
146      *
147      * @param x x-koordinaatin arvo
148      * @param y y-koordinaatin arvo
149      * @return tieto, ovatko koordinaatit alle alimman boundsin jolla pelaaja
150      * voi olla.
151      */
152     public boolean lowerBoundsCheck(int x, int y) {
153         if (x < 1) {
154             return true;
155         }
156         if (y < 1) {
157             return true;
158         }
159         return false;
160     }
161 
162     /**
163      * Varmistaa että halutut koordinaatit ovat pienemmät kuin niiden suurin
164      * mahdollinen arvo.
165      *
166      * @param x x-koordinaatin arvo
167      * @param y y-koordinaatin arvo
168      * @return tieto, ovatko koordinaatit yli suurimman boundsin jolla pelaaja
169      * voi olla.
170      */
171     public boolean upperBoundsCheck(int x, int y) {
172         if (x > this.alue.getBoundsX() - 1) {
173             return true;
174         }
175         if (y > this.alue.getBoundsY() - 1) {
176             return true;
177         }
178         return false;
179     }
180 
181     /**
182      * Tarkistaa voiko pelaaja liikkua käskyn mukaiseen koordinaattiin.
183      *
184      * @param x x-koordinaatin arvo jolle pelaaja haluaa siirtyä
185      * @param y y-koordinaatin arvo jolle pelaaja haluaa siirtyä
186      * @return onko parametrien koordinaatissa yhteentörmäystä vai ei
187      */
188     public boolean collision(int x, int y) {
189         for (Laatta laatta : this.laatat) {
190             if (x == laatta.getX() && y == laatta.getY()) {
191                 if (getPlayerSet()) {
192                     this.tk.kasitteleToiminto(laatta);
193                 }
194                 return laatta.collision();
195             }
196         }
197         return false;
198     }
199

201         return this.laatat;
202     }

204     public Tapahtumankasittelija getTK() {
205         return this.tk;
206     }
207     
208     public Pelaaja getPelaaja() {
209         return this.pelaaja;
210     }
